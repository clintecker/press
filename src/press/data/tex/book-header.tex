% The figure list prints as a Victorian "List of Plates". The book's
% contents show chapters only (tocdepth 0), which would also suppress the
% depth-1 figure lines, so the list file raises the depth for itself.
\AddToHook{begindocument/end}{%
  \renewcommand*{\listfigurename}{List of plates}%
  \addtocontents{lof}{\protect\setcounter{tocdepth}{1}}%
}

% A figure taller than the text block makes LuaLaTeX ship empty pages
% forever, silently. Rebind pandoc's image bounding with caption headroom:
% cap at \PressFigureCap tall, never wider than the line, never upscaled.
% Never raise the cap near \textheight. The cap defaults to the v1 6.3in
% here and is re-set per print profile (build/profile-geometry.tex), which
% is included after this header.
\providecommand{\PressFigureCap}{6.3in}
\makeatletter
\AtBeginDocument{%
  \ifdefined\pandocbounded
    \renewcommand*\pandocbounded[1]{%
      \sbox\pandoc@box{#1}%
      \Gscale@div\@tempa{\PressFigureCap}{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
      \Gscale@div\@tempb{\linewidth}{\wd\pandoc@box}%
      \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi
      \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
      \else\usebox{\pandoc@box}%
      \fi}%
  \else
    \providecommand{\maxwidth}{\linewidth}%
    \setkeys{Gin}{width=\maxwidth,height=\PressFigureCap,keepaspectratio}%
  \fi}
\makeatother

% Plain title page driven by document metadata (books that want the
% treatise front matter include tex/title-page.tex after this file).
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \raggedright
    \vspace*{0.8in}
    {\fontsize{25}{29}\selectfont\bfseries \@title\par}
    \vspace{0.35in}
    {\Large\color{muted} \@subtitle\par}
    \vfill
    {\large \@author\par}
    \vspace{0.12in}
    {\small\color{muted} \@date\par}
    \vspace{0.45in}
    {\small\color{muted} \directlua{tex.sprint(-2, os.getenv('BOOK_PUBLISHER') or '')}\par}
  \end{titlepage}
}
\makeatother
